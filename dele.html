Depois cole a chamada no final da sua gerarCertidao(id, qual) conforme indicado. // 1) Inicialização (escolha A ou B) const supabase = window .supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY); // ou, se precisar global: // window.deprecataSupabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY); // 2) Helpers obrigatórios (qual + object_path) function padronizaQualDoc ( qualAtual ) { const q = (qualAtual ?? "" ).toString().trim().toLowerCase(); // qual vindo das ações (decurso/permanencia/reiteracao) -> "certidoes" if ([ "decurso" , "permanencia" , "reiteracao" ].includes(q)) return "certidoes" ; if ([ "certidoes" , "comunicacoes" , "relatorios" , "outros" ].includes(q)) return q; return "outros" ; } function montarObjectPathDocx ( carta_id, qual, nome_arquivo_docx ) { const cId = (carta_id ?? "" ).toString().trim(); const q = (qual ?? "outros" ).toString().trim().toLowerCase(); const n = (nome_arquivo_docx ?? "" ).toString().trim().toLowerCase().endsWith( ".docx" ) ? (nome_arquivo_docx ?? "" ).toString().trim() : ` ${(nome_arquivo_docx ?? "" ).toString().trim()} .docx` ; return `docx/ ${cId} / ${q} / ${n} ` ; } // 3) Upload + insert (Storage -> public.carta_anexos) async function uploadAnexoDocx ( carta_id, qualAtual, blob, nome_arquivo_docx ) { const supa = window .deprecataSupabase || supabase; if (!supa) throw new Error ( "Supabase client não inicializado." ); const bucket = "deprecata-anexos" ; const qual = padronizaQualDoc(qualAtual); const object_path = montarObjectPathDocx(carta_id, qual, nome_arquivo_docx); const fileLike = blob instanceof File ? blob : new File([blob], nome_arquivo_docx, { type : "application/vnd.openxmlformats-officedocument.wordprocessingml.document" , }); // upload em Storage const { error : storageError } = await supa.storage .from(bucket) .upload(object_path, fileLike, { upsert : true }); if (storageError) throw storageError; // insert em public.carta_anexos (colunas/nomes exatos) const payload = { carta_id, qual, object_path, tipo_arquivo : "docx" , size_bytes : Number (fileLike.size ?? 0 ), nome_arquivo : nome_arquivo_docx, criado_em : new Date ().toISOString(), }; const { error : insertError } = await supa .from( "carta_anexos" ) .insert(payload); if (insertError) throw insertError; return { object_path, qual, nome_arquivo : nome_arquivo_docx }; } Como usar no final do seu gerarCertidao(id, qual) No seu gerarCertidao, depois de montar nome_arquivo_docx e gerar o DOCX: garanta que sua baixarDocx(...) retorna o blob chame uploadAnexoDocx(...)
